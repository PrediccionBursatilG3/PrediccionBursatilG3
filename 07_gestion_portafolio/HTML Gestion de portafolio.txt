<script src="https://cdnjs.cloudflare.com/ajax/libs/Chart.js/4.4.1/chart.umd.js"></script>
<script>
const donutData = [
  { label:'AAPL', pct:40, color:'#1F3864' },
  { label:'MSFT', pct:25, color:'#C5961A' },
  { label:'GOOGL', pct:20, color:'#378ADD' },
  { label:'BTC-USD', pct:15, color:'#B4B2A9' }
];

new Chart(document.getElementById('donutChart'), {
  type: 'doughnut',
  data: {
    labels: donutData.map(d=>d.label),
    datasets: [{ data: donutData.map(d=>d.pct), backgroundColor: donutData.map(d=>d.color), borderWidth: 3, borderColor: '#fff', hoverOffset: 4 }]
  },
  options: {
    responsive: true, maintainAspectRatio: false, cutout: '70%',
    plugins: { legend: { display: false }, tooltip: { callbacks: { label: ctx => ctx.label + ': ' + ctx.parsed + '%' } } }
  }
});

document.getElementById('donutLegend').innerHTML = donutData.map(d=>`
  <div class="leg-row">
    <span class="leg-left"><span class="leg-dot" style="background:${d.color}"></span>${d.label}</span>
    <span class="leg-pct">${d.pct}%</span>
  </div>`).join('');

const months = ['Jan 24','Mar 24','May 24','Jul 24','Sep 24','Nov 24','Jan 25','Mar 25','May 25','Jul 25','Sep 25','Nov 25','Jan 26','Mar 26','May 26'];
const strat  = [100,104,109,107,115,122,118,128,134,131,141,149,145,158,168];
const bhold  = [100,103,107,105,110,115,112,119,123,121,128,134,131,140,147];

new Chart(document.getElementById('equityChart'), {
  type: 'line',
  data: {
    labels: months,
    datasets: [
      { label:'Active strategy', data: strat, borderColor:'#1F3864', borderWidth: 2.5, pointRadius: 0, tension: 0.35, fill: false },
      { label:'Buy & Hold', data: bhold, borderColor:'#9ca3af', borderWidth: 1.5, borderDash: [5,4], pointRadius: 0, tension: 0.35, fill: false }
    ]
  },
  options: {
    responsive: true, maintainAspectRatio: false,
    plugins: { legend: { display: false }, tooltip: { mode:'index', intersect:false, callbacks:{ label: ctx => ctx.dataset.label+': '+ctx.parsed.y.toFixed(1) } } },
    scales: {
      x: { grid: { display: false }, ticks: { font:{size:11}, color:'#9ca3af', maxRotation:0, autoSkip:true, maxTicksLimit:7 } },
      y: { grid: { color:'#f3f4f6' }, ticks: { font:{size:11}, color:'#9ca3af', callback: v => v.toFixed(0) } }
    }
  }
});

const positions = [
  { ticker:'AAPL', icon:'AAPL', qty:120, buy:162.40, cur:218.75, signal:'BUY' },
  { ticker:'MSFT', icon:'MSFT', qty:85,  buy:310.20, cur:428.90, signal:'BUY' },
  { ticker:'GOOGL', icon:'GOOG', qty:60, buy:138.50, cur:192.30, signal:'HOLD' },
  { ticker:'BTC-USD', icon:'BTC', qty:1.4, buy:42800, cur:67420, signal:'BUY' },
  { ticker:'TSLA', icon:'TSLA', qty:95, buy:195.80, cur:163.40, signal:'SELL' },
  { ticker:'NVDA', icon:'NVDA', qty:40, buy:490.20, cur:1124.50, signal:'BUY' },
];

function fmt(n, dec=2) { return n.toLocaleString('en-US', {minimumFractionDigits:dec, maximumFractionDigits:dec}); }

document.getElementById('positionsBody').innerHTML = positions.map(p=>{
  const pnlDollar = (p.cur - p.buy) * p.qty;
  const pnlPct = ((p.cur - p.buy) / p.buy * 100);
  const pos = pnlDollar >= 0;
  const pillClass = p.signal==='BUY' ? 'pill-buy' : p.signal==='SELL' ? 'pill-sell' : 'pill-hold';
  return `<tr>
    <td><div class="ticker-cell"><div class="ticker-icon">${p.icon.slice(0,3)}</div>${p.ticker}</div></td>
    <td style="color:var(--color-text-secondary)">${p.qty}</td>
    <td style="color:var(--color-text-secondary)">$${fmt(p.buy)}</td>
    <td style="color:var(--color-text-primary)">$${fmt(p.cur)}</td>
    <td class="${pos?'pnl-pos':'pnl-neg'}">${pos?'+':''}\$${fmt(pnlDollar)}</td>
    <td class="${pos?'pnl-pos':'pnl-neg'}">${pos?'+':''}${fmt(pnlPct)}%</td>
    <td><span class="pill ${pillClass}">${p.signal}</span></td>
  </tr>`;
}).join('');
</script>
